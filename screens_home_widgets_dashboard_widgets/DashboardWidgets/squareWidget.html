static Widget squareWidget(
  context, {
  required String title,
  required Widget nextPage,
  required RxInt counting,
  required String image,
  required RxMap<String, int> dayMonthCount,
}) {
  final screenWidth = MediaQuery.of(context).size.width;

  return InkWell(
    onTap: () {
      if (title != 'Listed Vehicles') {
        Get.to(() => nextPage);
      }
    },
    mouseCursor: title == 'Listed Vehicles' ? MouseCursor.uncontrolled : null,
    child: Container(
      height: 150,
      width: screenWidth * 0.25,
      decoration: BoxDecoration(
          color: ColorManger.primaryColor,
          borderRadius: BorderRadius.circular(15)),
      child: Padding(
        padding: const EdgeInsets.all(15),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Row(
              children: [
                Container(
                  decoration: BoxDecoration(
                    color: Colors.orange,
                    borderRadius: BorderRadius.circular(10),
                  ),
                  padding: const EdgeInsets.all(5),
                  child: ImageIcon(
                    AssetImage(
                      image,
                    ),
                    color: Colors.white,
                  ),
                ),
                const SizedBox(
                  width: 10,
                ),
                Text(
                  title,
                  style: const TextStyle(
                    color: Colors.white,
                  ),
                ),
                const Spacer(),
                title != 'Listed Vehicles'
                    ? Container(
                        decoration: BoxDecoration(
                          color: Colors.orange,
                          borderRadius: BorderRadius.circular(5),
                        ),
                        padding: const EdgeInsets.all(5),
                        child: const ImageIcon(
                          AssetImage(AppAssets.arrowIcon),
                          color: Colors.black,
                        ),
                      )
                    : const SizedBox(),
              ],
            ),
            const SizedBox(
              height: 10,
            ),
            Obx(
              () => Text(
                counting.toString(),
                style: const TextStyle(color: Colors.white, fontSize: 30),
              ),
            ),
            const SizedBox(
              height: 20,
            ),
            Obx(() {
              return Row(
                children: [
                  Text(
                    'Today -  ${dayMonthCount['dayCount']}',
                    style: const TextStyle(color: Colors.white, fontSize: 13),
                  ),
                  const Spacer(),
                  Text(
                    'This Month - ${dayMonthCount['monthCount']}',
                    style: const TextStyle(color: Colors.white, fontSize: 13),
                  )
                ],
              );
            }),
          ],
        ),
      ),
    ),
  );
}
