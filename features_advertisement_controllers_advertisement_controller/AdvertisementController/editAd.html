Future<void> editAd(
  AdvertisementModel ad,
  String name,
  String category,
  String location,
  int hits,
  String adStatus,
  String attachment,
) async {
  final data = {
    "name": name,
    "location": location,
    "category": category,
    "hits": hits,
    "adStatus": adStatus,
    "attachment": attachment,
  };
  try {
    final response = await DioFactory.putdata(
      url: EndPoints.advertisement + ad.code!,
      data: data,
    );

    if (response.statusCode == 200) {
      Get.showSnackbar(
        const GetSnackBar(
          message: 'Advertisement edited successfully',
          duration: Duration(seconds: 2),
        ),
      );
      advertisementsList.clear();
      Get.to(const Advertisement());
      Get.showOverlay(
        asyncFunction: () => getAds(),
        loadingWidget: Center(
          child: CircularProgressIndicator(
            color: ColorManger.primaryColor,
          ),
        ),
      );
    } else {
      Get.showSnackbar(
        const GetSnackBar(
          message: 'An error has occurred',
          duration: Duration(seconds: 2),
        ),
      );
    }
  } catch (error) {
    Get.showSnackbar(
      GetSnackBar(
        message: 'An error has occurred: $error',
        duration: const Duration(seconds: 2),
      ),
    );
  }
}
