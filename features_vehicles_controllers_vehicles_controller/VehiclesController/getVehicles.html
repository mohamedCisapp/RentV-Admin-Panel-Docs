Future<void> getVehicles({int currentPage = 1}) async {
  try {
    Map data = {};
    data['status'] = "1";
    data['perPageRecords'] = 50;
    data['pageNo'] = currentPage;
    showPageNumber.value = true;

    if (filterMap['vehicleType'] != null) {
      data['vehicleType'] = filterMap['vehicleType'];
      data['pageNo'] = 1;
      data['perPageRecords'] = 1000;
      showPageNumber.value = false;
      vehicles.clear();
    }
    if (filterMap['fuelType'] != null) {
      data['fuelType'] = filterMap['fuelType'];
      data['pageNo'] = 1;
      data['perPageRecords'] = 1000;
      showPageNumber.value = false;
      vehicles.clear();
    }
    if (filterMap['seater'] != null) {
      data['seater'] = filterMap['seater'];
      data['pageNo'] = 1;
      data['perPageRecords'] = 1000;
      showPageNumber.value = false;
      vehicles.clear();
    }
    if (filterMap['transmission'] != null) {
      data['transmission'] = filterMap['transmission'];
      data['pageNo'] = 1;
      data['perPageRecords'] = 1000;
      showPageNumber.value = false;
      vehicles.clear();
    }
    if (filterMap['name'] != null) {
      data['name'] = filterMap['name'];
      data['pageNo'] = 1;
      data['perPageRecords'] = 1000;
      showPageNumber.value = false;
      vehicles.clear();
    }
    if (filterMap['email'] != null) {
      data['email'] = filterMap['email'];
      data['pageNo'] = 1;
      data['perPageRecords'] = 1000;
      showPageNumber.value = false;
      vehicles.clear();
    }
    if (filterMap['mobileNo'] != null) {
      data['mobileNo'] = filterMap['mobileNo'];
      data['pageNo'] = 1;
      data['perPageRecords'] = 1000;
      showPageNumber.value = false;
      vehicles.clear();
    }

    final response =
        await DioFactory.postdata(url: EndPoints.filterVehicles, data: data);

    if (response.statusCode == 200) {
      final jsonResponse = response.data;

      List<dynamic> data = jsonResponse["data"]["filteredData"];
      if (vehicles.isEmpty) {
        if (jsonResponse["data"]["totalCurrentRegisteredVehiclesValue"] !=
            null) {
          totalRecords.value =
              jsonResponse["data"]["totalCurrentRegisteredVehiclesValue"];
        }
        if (jsonResponse["data"]["totalRegisteredVehicles"] != null) {
          Get.put(HomrController()).registeredVehiclesCount.value =
              jsonResponse["data"]["totalRegisteredVehicles"];
        }

        if (jsonResponse["data"]["totalCurrentListedVehiclesValue"] != null) {
          Get.put(HomrController()).counting[0].value =
              jsonResponse["data"]["totalCurrentListedVehiclesValue"];
        }

        vehicles.clear();

        if (showPageNumber.isTrue) {
          vehicles.value = List.filled(totalRecords.value, FilterVehicle(),
              growable: true);
        }
      }
      if (showPageNumber.isFalse) {
        vehicles.addAll(data.map((item) => FilterVehicle.fromJson(item)));
      } else if (data.length < 50) {
        vehicles.removeRange(
            (currentPage - 1) * 50, (currentPage - 1) * 50 + data.length);
        vehicles.insertAll((currentPage - 1) * 50,
            data.map((item) => FilterVehicle.fromJson(item)));
      } else {
        vehicles.removeRange((currentPage - 1) * 50, currentPage * 50);
        vehicles.insertAll((currentPage - 1) * 50,
            data.map((item) => FilterVehicle.fromJson(item)));
      }
      if (kDebugMode) {
        print('GET Request Successful');
      }
    } else {
      if (kDebugMode) {
        print('GET Request Failed');
      }
      Get.showSnackbar(
        const GetSnackBar(
          message: 'An error has occurred',
          duration: Duration(seconds: 2),
        ),
      );
    }
  } catch (error) {
    if (kDebugMode) {
      print('Error: $error');
    }
  }
}
