factory VehicleModel.fromMap(Map<String, dynamic> map) {
  return VehicleModel(
    id: map['id']?.toInt(),
    code: map['code'],
    dateCreated: map['dateCreated']?.toInt(),
    createdBy: map['createdBy'],
    dateModified: map['dateModified']?.toInt(),
    modifiedBy: map['modifiedBy'],
    jsonData: map['json'] != null ? Json.fromMap(map['json']) : null,
    status: map['status'],
    ownerCode: map['ownerCode'],
    vehicleType: map['vehicleType'],
    modelName: map['modelName'],
    company: map['company'],
    modelCode: map['modelCode'],
    fuelType: map['fuelType'],
    registrationNumber: map['registrationNumber'],
    seater: map['seater']?.toInt(),
    rcDocument: map['rcDocument'] != null
        ? RcDocument.fromMap(map['rcDocument'])
        : null,
    transmission: map['transmission'],
  );
}
