void previousPage() async {
  if (currentPage.value == 1) {
    if (trips[(totalRecords / 50).floor() * 50] == TripModel()) {
      await Get.showOverlay(
        asyncFunction: () {
          return getTrips(currentPage: (totalRecords / 50).ceil());
        },
        loadingWidget: Center(
          child: CircularProgressIndicator(
            color: ColorManger.primaryColor,
          ),
        ),
      );
    }

    currentPage.value = (totalRecords / 50).ceil();
  } else {
    if (trips[(currentPage.value - 1) * 50] == TripModel()) {
      await Get.showOverlay(
        asyncFunction: () {
          return getTrips(currentPage: currentPage.value - 1);
        },
        loadingWidget: Center(
          child: CircularProgressIndicator(
            color: ColorManger.primaryColor,
          ),
        ),
      );
    }

    currentPage.value--;
  }
}
